model{
    # Priors:
    beta0 ~ dnorm(0, 0.01) # intercept
    beta1 ~ dnorm(0, 0.01) # slope
    sigma ~ dnorm(0, 100)I(0,) # standard deviation
    tau <- 1 / (sigma * sigma) # tau is precision (1 / variance)
    
    # Likelihood:
    for (i in 1:n){
      y[i] ~ dnorm(mu[i], tau) 
      mu[i] <- beta0 + beta1 * x[i]
    }
}
